CREATE TABLE PERSON (driverID VARCHAR(10),
name VARCHAR(10) NOT NULL,address VARCHAR(30),
PRIMARY KEY(driverID));


CREATE TABLE CAR (regno VARCHAR(10),
model VARCHAR(10), year INTEGER,
driverID VARCHAR(10) NOT NULL, PRIMARY KEY(regno),
FOREIGN KEY(driverID) REFERENCES PERSON(driverID) ON UPDATE CASCADE ON DELETE CASCADE);


CREATE TABLE ACCIDENT (report_number INTEGER,
acc_date DATE, location VARCHAR(30),
PRIMARY KEY(report_number));


CREATE TABLE PARTICIPATED (driverID VARCHAR(10) NOT NULL,
regno VARCHAR(10) NOT NULL, report_number INTEGER NOT NULL,
damage_amount INTEGER, PRIMARY KEY(driverID, regno, report_number),
FOREIGN KEY(driverID) REFERENCES PERSON(driverID) ON UPDATE CASCADE ON DELETE CASCADE,
FOREIGN KEY(regno) REFERENCES CAR(regno) ON UPDATE CASCADE ON DELETE CASCADE,
FOREIGN KEY(report_number) REFERENCES ACCIDENT(report_number) ON UPDATE CASCADE ON DELETE CASCADE);

ORDER PROCESSING DATABASE



CREATE TABLE CUSTOMER (custID INTEGER PRIMARY KEY,
cname VARCHAR(20) NOT NULL, city VARCHAR(20));



CREATE TABLE 'ORDER' (orderID INTEGER PRIMARY KEY,
odate DATE, custID INTEGER NOT NULL, order_amount INTEGER,
FOREIGN KEY(custID) REFERENCES CUSTOMER(custID) ON DELETE CASCADE ON UPDATE CASCADE);



CREATE TABLE ITEM(itemID INTEGER PRIMARY KEY,
unit_price INTEGER);



CREATE TABLE 'ORDER-ITEM'(orderID INTEGER NOT NULL,
itemID INTEGER NOT NULL, quantity INTEGER,
PRIMARY KEY(orderID,itemID),
FOREIGN KEY(orderID) REFERENCES 'ORDER'(orderID) ON DELETE CASCADE ON UPDATE CASCADE,
FOREIGN KEY(itemID) REFERENCES ITEM(itemID) ON DELETE CASCADE ON UPDATE CASCADE); 



CREATE TABLE WAREHOUSE(warehouseID INTEGER PRIMARY KEY,
city VARCHAR(20));



CREATE TABLE SHIPMENT(orderID INTEGER NOT NULL,
warehouseID INTEGER NOT NULL, ship_date DATE,
PRIMARY KEY(orderID,warehouseID),
FOREIGN KEY(orderID) REFERENCES 'ORDER'(orderID) ON UPDATE CASCADE ON DELETE CASCADE,
FOREIGN KEY(warehouseID) REFERENCES WAREHOUSE(werehouseID) ON DELETE CASCADE ON UPDATE CASCADE);

STUDENT ENROLLMENT




CREATE TABLE STUDENT(regno VARCHAR(10) PRIMARY KEY,
name VARCHAR(20) NOT NULL, major VARCHAR(5), bdate DATE);




CREATE TABLE COURSE(courseID INTEGER PRIMARY KEY,
cname VARCHAR(10) NOT NULL, dept VARCHAR(5));



CREATE TABLE 'TEXT'(book_ISBN INTEGER PRIMARY KEY,
book_title VARCHAR(10) NOT NULL, author VARCHAR(10),
publisher VARCHAR(10));



CREATE TABLE ENROLL (regno VARCHAR(10) NOT NULL,
courseID INTEGER NOT NULL, sem INTEGER, marks INTEGER,
PRIMARY KEY(regno,courseID),
FOREIGN KEY(regno) REFERENCES STUDENT(regno) ON DELETE CASCADE ON UPDATE CASCADE,
FOREIGN KEY(courseID) REFERENCES COURSE(courseID) ON DELETE CASCADE ON UPDATE CASCADE);




CREATE TABLE 'BOOK-ADOPTION'(courseID INTEGER NOT NULL,
book_ISBN INTEGER NOT NULL, sem INTEGER,
PRIMARY KEY(courseID,book_ISBN),
FOREIGN KEY(courseID) REFERENCES COURSE(courseID) ON DELETE CASCADE ON UPDATE CASCADE,
FOREIGN KEY(book_ISBN) REFERENCES 'TEXT'(book_ISBN) ON DELETE CASCADE ON UPDATE CASCADE);
